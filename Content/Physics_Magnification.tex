
Magnification lateral/heigth/size $y=z$, longitudinal/distance $x \perp y$

\vspace{1ex} \noindent
$ g \equiv $ distance from object, sender source (from Objectiv)
\newline
$ b \equiv $ distance from observer, receiver detector (from Okular)
\newline
$ l \equiv $ distance observer to next lense
\newline
$ r \equiv $ radius of curvature (bended reflective surface, lense koncav vs. konvex)
\newline
$ d \equiv $ max. thickness of curved lense enclosed between one konvex side from $g$ and one konkav side from $b$ with radi $r_g$ and $r_b$
\newline
$ s_0 \equiv $ sight/tubus length
\newline
$ \varepsilon \equiv $ "visual angle" from observer, $ \varepsilon_0 \equiv $ smallest visual angle
\newline
$ f \equiv $ focal length


\vspace{5ex} \noindent
$
g \parallel x \parallel b \parallel f 
\qquad
y_b \perp b	
\qquad	
y_g \perp g
$

\vspace{2ex} \noindent
$
\tan(\varepsilon) = \frac{y_b}{b}
\qquad
\tan(\varepsilon_0) = \frac{y_b}{g}
$

\vspace{2ex} \noindent
$
(\Delta\varphi)_{min} = \Delta\theta = 2 \varepsilon_0 = 1.22 \frac{\lambda}{d_y} 
$

\vspace{2ex} \noindent
"psychological visual angle":
$
2 \varepsilon_0^{min} = 1' = 60'' = \frac{1^{\circ}}{60} = \frac{2\pi}{360} \frac{1}{60} = \frac{\pi}{10800}
$

\vspace{2ex} \noindent
konkav: $ f>0, r<0 $	\qquad	konvex: $ f<0, r>0 $, virtual $ b<0 $


\vspace{5ex} \noindent
$
\frac{\eta_g}{g} + \frac{\eta_b}{b} = 
\frac{\eta - \eta_g}{r_g} - \frac{\eta - \eta_b}{r_b}
$
\newline \vspace{2ex}
$
\frac{1}{f_b} = \frac{1}{\eta_b} \left( \frac{\eta - \eta_g}{r_g} - \frac{\eta - \eta_b}{r_b} \right)
$
\newline \vspace{2ex}
$
\frac{1}{f_g} = \frac{1}{\eta_g} \left( \frac{\eta - \eta_g}{r_g} - \frac{\eta - \eta_b}{r_b} \right)
$
\newline \vspace{2ex}

$
\textnormal{thin lense: } f = f_g = f_b \iff d \rightarrow 0
$
\newline \vspace{2ex}

$
\textnormal{thik lense: } f = f_g = f_b \iff \eta_g = \eta_b 
$ \footnote{ $\eta_g = \eta_b = 1$ in vacuum ($\approx 1$ in air), not for eye }



\vspace{5ex}
spherical surface (panel), paraxial rays (linear approx., Gauss):
\vspace{2ex} \newline \noindent
$
\frac{\eta_g}{g} + \frac{\eta_b}{b} = 
\frac{\eta_b - \eta_g}{r}
$
\vspace{2ex} \newline \noindent
$
b = f_b = \frac{\eta_b}{\eta_b - \eta_g} \cdot r	\qquad
g = f_g = 
\frac{\eta_g}{\eta_b - \eta_g} \cdot r
$


\vspace{5ex}
Focal Length:
\vspace{2ex} \newline
$
f = 
\left( \frac{\Delta y}{\Delta \varepsilon} \right)_{min} = 
\frac{1}{r} \frac{\eta}{\Delta\eta} = 
\frac{r}{2} \frac{1}{\eta - 1} = 
- \frac{r}{2} = 
f_b = f_g
\stackrel{conjungtion}{=}
\frac{g \cdot b}{g + b}
$
\vspace{2ex} \newline
where: $ f^2 = x_g x_b $ and \newline
$ x_g = x_g(0) = g - f = -f \frac{y_g}{y_b} = +f \cdot \mathscr{G}_y $ and \newline 
$ x_b = x_b(0) = b - f = -f \frac{y_b}{y_g} = -f \cdot \mathscr{G}_y $ 


\vspace{5ex}
lense proportions:
\vspace{2ex} \newline \noindent
$
x_g(r_b) = -f \cdot \frac{\eta - 1}{\eta} \cdot \frac{d}{r_b}
\qquad
x_b(r_g) = -f \cdot \frac{\eta - 1}{\eta} \cdot \frac{d}{r_g}
$
\vspace{2ex} \newline \noindent
$
d = x_g(r_b) + x_b(r_g) + \Delta x
$
\qquad
where rays parallel within $\Delta x$ inside lense


\vspace{5ex}
Dioptri (Refractive Power)\footnote{Brechkraft}:
\vspace{2ex} \newline
$
D = 
\frac{\eta_g}{f_g} = 
\frac{\eta_b}{f_b}
\stackrel{vacuum}{=}
\frac{1}{f} = 
\frac{1}{g} + \frac{1}{b} 
\stackrel{paraxial}{=}
\frac{\Delta\eta}{f} =
\frac{2}{r}
\stackrel{concave}{=}
- \frac{2}{r} 
\stackrel{r_b = -r_g}{=}
\frac{2}{r} (\eta - 1) = 
(\eta - 1) \left( \frac{1}{r_g} - \frac{1}{r_b} + \frac{(\eta - 1) \cdot d}{\eta} \frac{d}{r_g \cdot r_b } \right)
\stackrel{conjungtion}{=}
\frac{g + b}{g \cdot b}
$ \qquad where: $ r < 0 $


\vspace{5ex}
longitudinal magnification:
\vspace{2ex} \newline
$
\mathscr{G}_x = \frac{g}{b}
$


\vspace{5ex}
lateral magnification:
\vspace{2ex} \newline
$
\mathscr{G}_y 
\stackrel{konkav}{=}
\frac{-y_b}{y_g} = 
- \frac{b}{g} 
\stackrel{konkvex}{=}
\frac{y_b}{y_g} = 
- \frac{b-r}{g+r}  = 
- \frac{f}{x_g} = 
- \frac{x_b}{f} = 
\frac{f}{f-g}
\stackrel{aplanar}{=}
\frac{\eta_g}{\eta_b} \cdot \frac{sin(u_g)}{sin(u_b)} 
\stackrel{kollinear}{=}
\frac{\eta_g}{\eta_b} \cdot \frac{tan(u_g)}{tan(u_b)} 
\stackrel{paraxial}{=}
\frac{\eta_g}{\eta_b} \cdot \frac{u_g}{u_b} 
$


\vspace{5ex}
viewangle magnification:
\vspace{2ex} \newline
$
\mathscr{G}_\varepsilon =
\frac{\varepsilon}{\varepsilon_0} =
\frac{\frac{y_b}{b+l}}{\frac{y_g}{s_0}} =
\frac{s_0}{g} \frac{1}{1+\frac{l}{b}} \cong
\frac{tan(\varepsilon)}{tan(\varepsilon_0)} =
\frac{y_b}{y_g} \cdot \frac{g}{b} = 
\mathscr{G}_y \cdot \mathscr{G}_x
\stackrel{astronom.}{=}
- \frac{f^{Obj}}{f^{Oku}}
\stackrel{terrestr.}{=}
+ \frac{f^{Obj}}{f^{Oku}}
\qquad
\in \left[ \frac{s_0}{f} , \frac{s_0}{f}+1 \right]
$


\vspace{5ex}
\textbf{total magnification:}
\vspace{2ex} \newline
$
\mathscr{G} = 
\mathscr{G}_y^2  \cdot \mathscr{G}_x = 
\mathscr{G}_y \cdot \frac{\varepsilon}{\varepsilon_0} = 
\mathscr{G}_y \cdot \mathscr{G}_\varepsilon = 
-\frac{x_b^{Obj}}{f^{Obj}} \cdot \frac{s_0}{f^{Oku}}
$
