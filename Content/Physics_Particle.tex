\textbf{Particle Density: Point and Wave}
\vspace{1ex} \newline 

\index{Particle}
Particles N $\equiv$ Natural Oscillations\footnote{Eingenschwingungen, intrinsic} of blackbody within a range of frequency spectrum\footnote{Frequenzintervall} $(\nu,\nu+d\nu)$ $\equiv$ \index{ocscillation, self}
Vibration Modes\footnote{Schwingungsmoden} $\equiv$ Points $\equiv$ exactly one of each possible wavevectors $\vec{k}$ \index{Point particle} \index{Wavevector}
\index{Mode}
\vspace{2ex} \newline 

\noindent
$a \equiv $ particle size (radius, extension) of each individual particle $i = \left[1,N\right]$
\vspace{2ex} \newline 
$\frac{\pi}{a} = $ Grid Constant $\equiv$ next point distance between those N points \index{Grid Constant}
\vspace{2ex} \newline 
$0 \neq m_i = \frac{2a}{\lambda_i} \in \mathbb{N}_+ \qquad \lambda = \frac{c}{\nu} = \frac{\omega}{k} \frac{1}{\nu}$
\vspace{2ex} \newline 
$ |\vec{k}| = k = \frac{\pi}{a} \sqrt{ \sum m_i^2 } = \frac{\omega}{c} = \frac{2\pi\nu}{c} = \frac{2\pi}{\lambda} \equiv $ Sphere Radius (containing the grid of particles inside) \index{Sphere Radius}
\vspace{2ex} \newline 
$\sum m_i^2 \gg 1 \iff k \gg \frac{\pi}{a} \iff \lambda \ll 2a$: N in blackbody independent of geometrical form
\vspace{2ex} \newline 
$
N = 
\frac{1}{3} \frac{1}{2\pi^2} \left( a \frac{\omega}{c} \right)^3 = 
\frac{1}{3} 2^2\pi \left( a \frac{\nu}{c} \right)^3 = 
\frac{4\pi}{3} \left( \frac{a}{\lambda} \right)^3
$
\vspace{2ex} \newline 
$
N(\omega) = 
\frac{1}{3} \frac{1}{\pi^2} \left( \frac{\omega \cdot a}{c} \right)^3
$
\vspace{2ex} \newline 
$
N(\nu) = 
\frac{1}{3} \cdot 2^2 \cdot 2\pi \left( \frac{\nu \cdot a}{c} \right)^3
$
\vspace{2ex} \newline 


Density: \index{Density}
\vspace{2ex} \newline 
$
n = \frac{N}{V} = \frac{N}{A^3} = 
\frac{8\pi}{3} \left( \frac{\nu}{c} \right)^3 = 
\frac{1}{3} \pi \left( \frac{2}{\lambda} \right)^3
$
\vspace{2ex} \newline 
$
n_{\nu} = 
\frac{dn}{d\nu} = 
8\pi \frac{\nu^2}{c^3} = 
3 \frac{n}{\nu} 
$
\vspace{2ex} \newline 
$
n_{\lambda} = 
- \frac{dn}{d\lambda} = 
8\pi \frac{1}{\lambda^4} = 
3 \frac{n}{\lambda}
$
\vspace{2ex} \newline
