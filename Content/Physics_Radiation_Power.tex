Power $\equiv$ \textbf{Radiant Flux}:\footnote{Strahlungsleistung, Strahlungsfluss, Energiestrom}
\index{Power}	\index{Flux}

$
\Phi = 
\frac{d}{dt}W = 
|\frac{1}{\mu_0}(\mathscr{E} \times \mathscr{B} ) | = 
| \mathscr{S} | = 
\oint\limits_A\, \mathscr{S}\, d\vec{A}
$

$dW \equiv$ Radiant Energy\footnote{Strahlungsenergie}, $W \equiv$ Work
\index{Energy}

$\mathscr{S} \equiv$ Poynting-Vector, $A \equiv$ radiant surface area, $d\vec{A} \perp dA$	\index{Poynting-Vector}


\vspace{5ex}
\textbf{Flux Density} $\equiv$ Radiant Intensity (ray beam angular from source at distance $g$):\footnote{Strahlstärke, Flussdichte}
\index{Flux Density} \index{Intensity}

$
\mathscr{J} = \frac{d\Phi}{d\Omega_g}
$

$
\mathscr{J}_{\lambda} = \frac{c}{\lambda^2} \mathscr{J}_{\nu}
$

$
d\mathscr{J} = \frac{d^2\Phi}{d\Omega_g} = 
\frac{d^2\Phi}{ \frac{dA_b \cos(\varepsilon_b)}{R^2} } = 
\mathscr{L} dA_g \cos(\varepsilon_g)
$
\newline where Radiance\footnote{experimentally measured} $\mathscr{L}$ = const.


\vspace{5ex}
\textbf{Irradiance} $\equiv$ Impact Intensity (ray beam straight at receiver in distance $b$):\footnote{Bestrahlungsstärke}
\index{Irradiance}

$
\mathscr{I} = \frac{d\Phi}{dA_b} = \mathscr{J} \frac{\cos(\varepsilon_b)}{R^2} = \frac{d\mathscr{H}}{dt}
$


\vspace{5ex}
\textbf{Radiant Exposure} $\equiv$ Action:\footnote{Bestrahlung, Strahlung-Wirkung auf Fläche A}
\index{Radiant Exposure}

$
\mathscr{H} = \int\limits_{t_1}^{t_2}\, \mathscr{I}(t)\, dt
$


\vspace{5ex}
\textbf{Radiant Exitance} $\equiv$ Radiant Emittance:\footnote{spezifische Ausstrahlung, Leuchtkraft}
\index{Exitance} \index{Emittance}

$
\mathscr{M} = \frac{d\Phi}{dA_g}
$


\vspace{5ex} 
\textbf{Radiance} $\equiv$ Radiant Density (Apparency, Brightness):\footnote{Strahldichte, Helligkeit, prop. Leistung, $\rightarrow$ Strahlungsfeld}
\index{Radiance} \index{Radiant Density}

\vspace{1ex} 
Emissity\footnote{Emissionsvermögen} $\equiv \mathscr{L}_g$ \index{Emissity}

Appearence Areas:  $dA_g \cos(\varepsilon_g)$ and $dA_b \cos(\varepsilon_b)$ \index{Appearence}
\vspace{1ex} 

$
\mathscr{L} = 
\frac{d\mathscr{J}}{dA_g \cdot \cos(\varepsilon_g)} = 
\frac{1}{dA_g \cos(\varepsilon_g)} \frac{d^2\Phi}{d\Omega_g} = 
\frac{1}{dA_b \cos(\varepsilon_b)} \frac{d^2\Phi}{d\Omega_b} = 
\frac{c}{4\pi} $w$_{EM} 
$

\vspace{2ex} 
$\mathscr{L} = $ const.:\footnote{"Lambert's cosine law"} diffused emission, reflection, brightness, rough no mirroring surface
\index{Lambert}
...

$\mathscr{L} \neq$ const.: radiation transfer, no diffusion, apparent, smooth surface mirroring
